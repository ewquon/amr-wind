59 // Populate normal vector with special treatment of overset boundary
60 void populate_normal_vector(
61  amrex::MultiFab& mf_normvec,
62  const amrex::MultiFab& mf_vof,
63  const amrex::iMultiFab& mf_iblank)
64 {
65  const auto& normvec = mf_normvec.arrays();
66  const auto& vof = mf_vof.const_arrays();
67  const auto& iblank = mf_iblank.const_arrays();
68  // Calculate gradients in each direction with centered diff
69  amrex::ParallelFor(
70  mf_normvec, mf_normvec.n_grow - amrex::IntVect(1),
71  [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
72  // Neumann condition across nalu bdy
73  int ibdy =
74  (iblank[nbx](i, j, k) != iblank[nbx](i - 1, j, k)) ? -1 : 0;
75  int jbdy =
76  (iblank[nbx](i, j, k) != iblank[nbx](i, j - 1, k)) ? -1 : 0;
77  int kbdy =
78  (iblank[nbx](i, j, k) != iblank[nbx](i, j, k - 1)) ? -1 : 0;
79  // no cell should be isolated such that -1 and 1 are needed
80  ibdy =
81  (iblank[nbx](i, j, k) != iblank[nbx](i + 1, j, k)) ? +1 : ibdy;
82  jbdy =
83  (iblank[nbx](i, j, k) != iblank[nbx](i, j + 1, k)) ? +1 : jbdy;
84  kbdy =
85  (iblank[nbx](i, j, k) != iblank[nbx](i, j, k + 1)) ? +1 : kbdy;
86  // Calculate normal
87  amrex::Real mx, my, mz, mmag;
88  multiphase::youngs_finite_difference_normal_neumann(
89  i, j, k, ibdy, jbdy, kbdy, vof[nbx], mx, my, mz);
90  // Normalize normal
91  mmag = std::sqrt(mx * mx + my * my + mz * mz + 1e-20);
92  // Save normal
93  normvec[nbx](i, j, k, 0) = mx / mmag;
94  normvec[nbx](i, j, k, 1) = my / mmag;

202 amrex::Real calculate_pseudo_dt_flux(
203  amrex::MultiFab& mf_fx,
204  amrex::MultiFab& mf_fy,
205  amrex::MultiFab& mf_fz,
206  amrex::MultiFab& mf_vof,
207  amrex::Real tol)
208 {
209  // Get the maximum flux magnitude, but just for vof fluxes
210  const amrex::Real pdt_fx = amrex::ReduceMin(
211  mf_fx, mf_vof, 0,
212  [=] AMREX_GPU_HOST_DEVICE(
213  amrex::Box const& bx, amrex::Array4<amrex::Real const> const& fx,
214  amrex::Array4<amrex::Real const> const& vof) -> amrex::Real {
215  amrex::Real pdt_fab = 1.0;
216  amrex::Loop(bx, [=, &pdt_fab](int i, int j, int k) noexcept {
217  amrex::Real pdt_lim = 1.0;
218  if (fx(i, j, k, 0) > tol && vof(i, j, k) > tol) {
219  // VOF is removed from cell i

222  // VOF is removed from cell i-1

226  });
227  return pdt_fab;
228  });
229  const amrex::Real pdt_fy = amrex::ReduceMin(
230  mf_fy, mf_vof, 0,
231  [=] AMREX_GPU_HOST_DEVICE(
232  amrex::Box const& bx, amrex::Array4<amrex::Real const> const& fy,
233  amrex::Array4<amrex::Real const> const& vof) -> amrex::Real {
234  amrex::Real pdt_fab = 1.0;
235  amrex::Loop(bx, [=, &pdt_fab](int i, int j, int k) noexcept {
236  amrex::Real pdt_lim = 1.0;
237  if (fy(i, j, k, 0) > tol && vof(i, j, k) > tol) {
238  // VOF is removed from cell j
239  pdt_lim = vof(i, j, k) / fy(i, j, k, 0);
240  } else if (fy(i, j, k, 0) < -tol && vof(i, j - 1, k) > tol) {
241  // VOF is removed from cell j-1
242  pdt_lim = vof(i, j - 1, k) / -fy(i, j, k, 0);
243  }
244  pdt_fab = amrex::min(pdt_fab, pdt_lim);
245  });
246  return pdt_fab;
247  });
248  const amrex::Real pdt_fz = amrex::ReduceMin(
249  mf_fz, mf_vof, 0,
250  [=] AMREX_GPU_HOST_DEVICE(
251  amrex::Box const& bx, amrex::Array4<amrex::Real const> const& fz,
252  amrex::Array4<amrex::Real const> const& vof) -> amrex::Real {
253  amrex::Real pdt_fab = 1.0;

255  amrex::Real pdt_lim = 1.0;
256  if (fz(i, j, k, 0) > tol && vof(i, j, k) > tol) {

259  } else if (fz(i, j, k, 0) < -tol && vof(i, j, k - 1) > tol) {
260  // VOF is removed from cell k-1
261  pdt_lim = vof(i, j, k - 1) / -fz(i, j, k, 0);
262  }
263  pdt_fab = amrex::min(pdt_fab, pdt_lim);
264  });
265  return pdt_fab;
266  });
267  const amrex::Real pdt = amrex::min(pdt_fx, amrex::min(pdt_fy, pdt_fz));
268  return pdt;
269 }

438 {

442  amrex::ParallelFor(
